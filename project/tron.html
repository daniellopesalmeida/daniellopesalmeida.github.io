Tron
Overview
Tron is a C++ game engine built from the ground up, using Game Programming Patterns book as reference.
Built from Minigin, which is an SDL2 + glm C++ framework from DAE (Digital Arts and Entertainment) during my studies, several core systems based on Game Programming Patterns such as Component pattern, Game Loop, Update Method, State Machine, Event/Observer system and a Service Locator for audio where implemented.
Architecture Patterns
The GameObject Component patterns allows to create flexible, reusable entity systems.
class GameObject final: public std::enable_shared_from_this<GameObject>
{
public:
    void Update(float deltaTime);
    void FixedUpdate();
    void Render() const;

    void SetPosition(float x, float y);
    TransformComponent* GetTransform() const { return m_Transform; }

    void SetParent(std::shared_ptr<GameObject> parent, bool keepWorldPos = false);
    GameObject* GetParent() const { return m_Parent; }
    const std::vector<std::shared_ptr<GameObject>>& GetChildren() { return m_pChildren; }

    // Component composition
    template <typename T, typename... TArgs>
    T* AddComponent(TArgs&&... args);

    template <typename T>
    T* GetComponent() const;
};
Components inherit from a base class. Each component encapsulates specific behavior independently, allowing GameObjects to be composed of any combination of components.
class Component
{
public:
    Component(GameObject* pOwner = nullptr);
    virtual ~Component() = default;

    Component(const Component&) = delete;
    Component(Component&&) noexcept = delete;
    Component& operator=(const Component&) = delete;

    virtual void Update(float deltaTime);
    virtual void Render();
    virtual void FixedUpdate();

    void Delete() { m_MarkedForDelete = true; }
    bool IsMarkedForDeletion() const { return m_MarkedForDelete; }

protected:
    GameObject* GetOwner() const { return m_pOwner; }

private:
    GameObject* m_pOwner;
    bool m_MarkedForDelete{ false };
};
A Singleton template provides the foundation for managing global instances, mostly known as managers.
template <typename T>
class Singleton
{
public:
    static T& GetInstance()
    {
        static T instance{};
        return instance;
    }

    virtual ~Singleton() = default;
    Singleton(const Singleton&) = delete;
    Singleton(Singleton&&) = delete;
    Singleton& operator=(const Singleton&) = delete;

protected:
    Singleton() = default;
};
The Service Locator manages system services like audio without being tightly coupled. It supports runtime registration and null implementations.
class ServiceLocator final
{
    static std::unique_ptr<SoundSystem> _ss_instance;
public:
    static SoundSystem& GetSoundSystem() 
    { 
        return *_ss_instance; 
    }

    static void RegisterSoundSystem(std::unique_ptr<SoundSystem>&& ss) 
    {
        _ss_instance = ss == nullptr ? 
            std::make_unique<NullSoundSystem>() : 
            std::move(ss);
    }
};
The State Machine pattern manages discrete behavioral states with OnEnter/OnExit transitions. Each state owns its logic independently while maintaining access to its owning GameObject.
class StateMachine
{
public:
    StateMachine() = default;
    virtual ~StateMachine() = default;

    StateMachine(const StateMachine&) = delete;
    StateMachine(StateMachine&&) = delete;

    virtual void OnEnter() = 0;
    virtual void OnExit() = 0;
    virtual void Update() = 0;

    GameObject* GetOwner() const { return m_pOwner; }
    void SetOwner(GameObject* owner) { m_pOwner = owner; }

private:
    GameObject* m_pOwner{};
};
